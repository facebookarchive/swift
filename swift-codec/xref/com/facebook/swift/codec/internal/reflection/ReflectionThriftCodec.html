45  @Immutable
46  public class ReflectionThriftCodec<T> implements ThriftCodec<T>
47  {
48      private final ThriftStructMetadata<T> metadata;
49      private final SortedMap<Short, ThriftCodec<?>> fields;

51      public ReflectionThriftCodec(ThriftCodecManager manager, ThriftStructMetadata<T> metadata)
52      {
53          this.metadata = metadata;
54          ImmutableSortedMap.Builder<Short, ThriftCodec<?>> fields = ImmutableSortedMap.naturalOrder();
55          for (ThriftFieldMetadata fieldMetadata : metadata.getFields()) {
56              fields.put(fieldMetadata.getId(), manager.getCodec(fieldMetadata.getType()));
57          }
58          this.fields = fields.build();
59      }

67      @Override
68      public T read(TProtocol protocol)
69              throws Exception
70      {
71          TProtocolReader reader = new TProtocolReader(protocol);
72          reader.readStructBegin();
73  
74          Map<Short, Object> data = new HashMap<>(metadata.getFields().size());
75          while (reader.nextField()) {
76              short fieldId = reader.getFieldId();
77  
78              // do we have a codec for this field
79              ThriftCodec<?> codec = fields.get(fieldId);
80              if (codec == null) {
81                  reader.skipFieldData();
82                  continue;
83              }
84  
85              // is this field readable
86              ThriftFieldMetadata field = metadata.getField(fieldId);
87              if (field.isWriteOnly()) {
88                  reader.skipFieldData();
89                  continue;
90              }
91  
92              // read the value
93              Object value = reader.readField(codec);
94              if (value == null) {
95                  continue;
96              }
97  
98              data.put(fieldId, value);
99          }
100         reader.readStructEnd();
101 
102         // build the struct
103         return constructStruct(data);
104     }

106     @Override
107     public void write(T instance, TProtocol protocol)
108             throws Exception
109     {
110         TProtocolWriter writer = new TProtocolWriter(protocol);
111         writer.writeStructBegin(metadata.getStructName());
112 
113         for (ThriftFieldMetadata fieldMetadata : metadata.getFields()) {
114             // is the field writable?
115             if (fieldMetadata.isReadOnly()) {
116                 continue;
117             }
118 
119             // get the field value
120             Object fieldValue = getFieldValue(instance, fieldMetadata);
121 
122             // write the field
123             if (fieldValue != null) {
124                 ThriftCodec<Object> codec = (ThriftCodec<Object>) fields.get(fieldMetadata.getId());
125                 writer.writeField(fieldMetadata.getName(), fieldMetadata.getId(), codec, fieldValue);
126             }
127         }

131     private T constructStruct(Map<Short, Object> data)
132             throws Exception
133     {
134         // construct instance
135         Object instance;
136         {
137             ThriftConstructorInjection constructor = metadata.getConstructor();
138             Object[] parametersValues = new Object[constructor.getParameters().size()];
139             for (ThriftParameterInjection parameter : constructor.getParameters()) {
140                 Object value = data.get(parameter.getId());
141                 parametersValues[parameter.getParameterIndex()] = value;
142             }
143 
144             try {
145                 instance = constructor.getConstructor().newInstance(parametersValues);
146             }
147             catch (InvocationTargetException e) {
148                 if (e.getTargetException() != null) {
149                     Throwables.propagateIfInstanceOf(e.getTargetException(), Exception.class);
150                 }
151                 throw e;
152             }
153         }
154 
155         // inject fields
156         for (ThriftFieldMetadata fieldMetadata : metadata.getFields()) {
157             for (ThriftInjection injection : fieldMetadata.getInjections()) {
158                 if (injection instanceof ThriftFieldInjection) {
159                     ThriftFieldInjection fieldInjection = (ThriftFieldInjection) injection;
160                     Object value = data.get(fieldInjection.getId());
161                     if (value != null) {
162                         fieldInjection.getField().set(instance, value);
163                     }
164                 }
165             }
166         }
167 
168         // inject methods
169         for (ThriftMethodInjection methodInjection : metadata.getMethodInjections()) {
170             boolean shouldInvoke = false;
171             Object[] parametersValues = new Object[methodInjection.getParameters().size()];
172             for (ThriftParameterInjection parameter : methodInjection.getParameters()) {
173                 Object value = data.get(parameter.getId());
174                 if (value != null) {
175                     parametersValues[parameter.getParameterIndex()] = value;
176                     shouldInvoke = true;
177                 }
178             }
179 
180             if (shouldInvoke) {
181                 try {
182                     methodInjection.getMethod().invoke(instance, parametersValues);
183                 }
184                 catch (InvocationTargetException e) {
185                     if (e.getTargetException() != null) {
186                         Throwables.propagateIfInstanceOf(e.getTargetException(), Exception.class);
187                     }
188                     throw e;
189                 }
190             }
191         }
192 
193         // builder method
194         if (metadata.getBuilderMethod() != null) {
195             ThriftMethodInjection builderMethod = metadata.getBuilderMethod();
196             Object[] parametersValues = new Object[builderMethod.getParameters().size()];
197             for (ThriftParameterInjection parameter : builderMethod.getParameters()) {
198                 Object value = data.get(parameter.getId());
199                 parametersValues[parameter.getParameterIndex()] = value;
200             }
201 
202             try {
203                 instance = builderMethod.getMethod().invoke(instance, parametersValues);
204                 if (instance == null) {
205                     throw new IllegalArgumentException("Builder method returned a null instance");
206 
207                 }
208                 if (!metadata.getStructClass().isInstance(instance)) {
209                     throw new IllegalArgumentException(format("Builder method returned instance of type %s, but an instance of %s is required",
210                             instance.getClass().getName(),
211                             metadata.getStructClass().getName()));
212                 }
213             }
214             catch (InvocationTargetException e) {
215                 if (e.getTargetException() != null) {
216                     Throwables.propagateIfInstanceOf(e.getTargetException(), Exception.class);
217                 }
218                 throw e;
219             }
220         }
221 
222         return (T) instance;
223     }

225     private Object getFieldValue(Object instance, ThriftFieldMetadata field)
226             throws Exception
227     {
228         try {
229             ThriftExtraction extraction = field.getExtraction();
230             Object value;
231             if (extraction instanceof ThriftFieldExtractor) {
232                 ThriftFieldExtractor thriftFieldExtractor = (ThriftFieldExtractor) extraction;
233                 value = thriftFieldExtractor.getField().get(instance);
234             }
235             else if (extraction instanceof ThriftMethodExtractor) {
236                 ThriftMethodExtractor thriftMethodExtractor = (ThriftMethodExtractor) extraction;
237                 value = thriftMethodExtractor.getMethod().invoke(instance);
238             }
239             else {
240                 throw new IllegalAccessException("Unsupported field extractor type " + extraction.getClass().getName());
241             }
242 
243             return value;
244         }
245         catch (InvocationTargetException e) {
246             if (e.getTargetException() != null) {
247                 Throwables.propagateIfInstanceOf(e.getTargetException(), Exception.class);
248             }
249             throw e;
250         }

94      public static <T extends Annotation> Set<T> getAllClassAnnotations(Class<?> type, Class<T> annotation)
95      {
96          // if the class is directly annotated, it is considered the only annotation
97          if (type.isAnnotationPresent(annotation)) {
98              return ImmutableSet.of(type.getAnnotation(annotation));
99          }
100 
101         // otherwise find all annotations from all super classes and interfaces
102         ImmutableSet.Builder<T> builder = ImmutableSet.builder();
103         addAllClassAnnotations(type, annotation, builder);
104         return builder.build();
105     }

107     private static <T extends Annotation> void addAllClassAnnotations(Class<?> type, Class<T> annotation, ImmutableSet.Builder<T> builder)
108     {
109         if (type.isAnnotationPresent(annotation)) {
110             builder.add(type.getAnnotation(annotation));
111         }
112         if (type.getSuperclass() != null) {
113             addAllClassAnnotations(type.getSuperclass(), annotation, builder);
114         }
115         for (Class<?> anInterface : type.getInterfaces()) {
116             addAllClassAnnotations(anInterface, annotation, builder);
117         }
118     }

120     public static Iterable<Method> getAllDeclaredMethods(Class<?> type)
121     {
122         ImmutableList.Builder<Method> methods = ImmutableList.builder();
123 
124         for (Class<?> clazz = type; (clazz != null) && !clazz.equals(Object.class); clazz = clazz.getSuperclass()) {
125             methods.addAll(ImmutableList.copyOf(clazz.getDeclaredMethods()));
126         }
127         return methods.build();
128     }

139     /**
140      * Find methods that are tagged with a given annotation somewhere in the hierarchy
141      */
142     public static Collection<Method> findAnnotatedMethods(Class<?> type, Class<? extends Annotation> annotation)
143     {
144         List<Method> result = new ArrayList<>();
145 
146         // gather all publicly available methods
147         // this returns everything, even if it's declared in a parent
148         for (Method method : type.getMethods()) {
149             // skip methods that are used internally by the vm for implementing covariance, etc
150             if (method.isSynthetic() || method.isBridge() || isStatic(method.getModifiers())) {
151                 continue;
152             }
153 
154             // look for annotations recursively in super-classes or interfaces
155             Method managedMethod = findAnnotatedMethod(
156                     type,
157                     annotation,
158                     method.getName(),
159                     method.getParameterTypes());
160             if (managedMethod != null) {
161                 result.add(managedMethod);
162             }
163         }
164 
165         return result;
166     }

168     @SuppressWarnings("PMD.EmptyCatchBlock")
169     public static Method findAnnotatedMethod(Class<?> configClass, Class<? extends Annotation> annotation, String methodName, Class<?>... paramTypes)
170     {
171         try {
172             Method method = configClass.getDeclaredMethod(methodName, paramTypes);
173             if (method != null && method.isAnnotationPresent(annotation)) {
174                 return method;
175             }
176         }
177         catch (NoSuchMethodException e) {
178             // ignore
179         }
180 
181         if (configClass.getSuperclass() != null) {
182             Method managedMethod = findAnnotatedMethod(
183                     configClass.getSuperclass(),
184                     annotation,
185                     methodName,
186                     paramTypes);
187             if (managedMethod != null) {
188                 return managedMethod;
189             }
190         }
191 
192         for (Class<?> iface : configClass.getInterfaces()) {
193             Method managedMethod = findAnnotatedMethod(iface, annotation, methodName, paramTypes);
194             if (managedMethod != null) {
195                 return managedMethod;
196             }
197         }
198 
199         return null;
200     }

202     public static Collection<Field> findAnnotatedFields(Class<?> type, Class<? extends Annotation> annotation)
203     {
204         List<Field> result = new ArrayList<>();
205 
206         // gather all publicly available methods
207         // this returns everything, even if it's declared in a parent
208         for (Field field : type.getFields()) {
209             if (field.isSynthetic() || isStatic(field.getModifiers())) {
210                 continue;
211             }
212 
213             if (field.isAnnotationPresent(annotation)) {
214                 result.add(field);
215             }
216         }
217 
218         return result;
219     }

233     private static class ThriftFieldParanamer extends AnnotationParanamer
234     {
235         @Override
236         protected String getNamedValue(Annotation annotation)
237         {
238             if (annotation instanceof ThriftField) {
239                 String name = ((ThriftField) annotation).name();
240                 if (!name.isEmpty()) {
241                     return name;
242                 }
243             }
244             return super.getNamedValue(annotation);
245         }
246 
247         @Override
248         protected boolean isNamed(Annotation annotation)
249         {
250             return (annotation instanceof ThriftField) || super.isNamed(annotation);
251         }

254     private static class GeneralParanamer implements Paranamer
255     {
256         @Override
257         public String[] lookupParameterNames(AccessibleObject methodOrConstructor)
258         {
259             String[] names;
260             if (methodOrConstructor instanceof Method) {
261                 Method method = (Method) methodOrConstructor;
262                 names = new String[method.getParameterTypes().length];
263             }
264             else if (methodOrConstructor instanceof Constructor<?>) {
265                 Constructor<?> constructor = (Constructor<?>) methodOrConstructor;
266                 names = new String[constructor.getParameterTypes().length];
267             }
268             else {
269                 throw new IllegalArgumentException("methodOrConstructor is not an instance of Method or Constructor but is " + methodOrConstructor.getClass().getName());
270             }
271             for (int i = 0; i < names.length; i++) {
272                 names[i] = "arg" + i;
273             }
274             return names;
275         }
276 
277         @Override
278         public String[] lookupParameterNames(AccessibleObject methodOrConstructor, boolean throwExceptionIfMissing)
279         {
280             return lookupParameterNames(methodOrConstructor);
281         }

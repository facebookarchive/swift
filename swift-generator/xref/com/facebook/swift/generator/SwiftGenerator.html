38  /**
39   * Parses a Thrift IDL file and writes out initial annotated java classes.
40   */
41  public class SwiftGenerator
42  {
43      private static final String THRIFT_FOLDER = System.getProperty("user.home") + "/fb/src/swift/swift-idl-parser/src/test/resources";
44      private static final String OUTPUT_FOLDER = System.getProperty("user.home") + "/fb/workspace/default/Demo/src/";
45  
46      public static void main(final String ... args) throws Exception
47      {
48          final SwiftGeneratorConfig config = new SwiftGeneratorConfig(new File(THRIFT_FOLDER),
49                                                                       new File [] { new File(THRIFT_FOLDER, "hive/metastore.thrift"),
50                                                                                     new File(THRIFT_FOLDER, "fb303.thrift") },
51                                                                       new File(OUTPUT_FOLDER),
52                                                                       "com.fb.test");
53          final SwiftGenerator generator = new SwiftGenerator(config);
54          generator.parse();
55      }
56  
57      private final File outputFolder;
58      private final SwiftGeneratorConfig swiftGeneratorConfig;
59  
60      private final TemplateLoader templateLoader;
61  
62      public SwiftGenerator(final SwiftGeneratorConfig swiftGeneratorConfig)
63      {
64          this.swiftGeneratorConfig = swiftGeneratorConfig;
65  
66          this.outputFolder = swiftGeneratorConfig.getOutputFolder();
67          if (outputFolder != null) {
68              outputFolder.mkdirs();
69          }
70  
71          this.templateLoader = new TemplateLoader("java/regular.st");
72      }
73  
74      public void parse() throws Exception
75      {
76          final List<SwiftDocumentContext> contexts = Lists.newArrayList();
77          for (final File file : swiftGeneratorConfig.getInputFiles()) {
78              contexts.add(parseDocument(file, new TypeRegistry()));
79          }
80  
81          for (final SwiftDocumentContext context : contexts) {
82              generateFiles(context);
83          }
84      }
85  
86      private SwiftDocumentContext parseDocument(final File thriftFile,
87                                                 final TypeRegistry typeRegistry) throws IOException
88      {
89          final String thriftName = thriftFile.getName();
90          final int idx = thriftName.lastIndexOf('.');
91          final String thriftNamespace = (idx == -1) ? thriftName : thriftName.substring(0, idx);
92  
93          Preconditions.checkState(thriftFile.exists(), "The file %s does not exist!", thriftFile.getAbsolutePath());
94          Preconditions.checkState(thriftFile.canRead(), "The file %s can not be read!", thriftFile.getAbsolutePath());
95          Preconditions.checkState(!StringUtils.isEmpty(thriftNamespace), "The file %s can not be translated to a namespace", thriftFile.getAbsolutePath());
96          final SwiftDocumentContext context = new SwiftDocumentContext(thriftFile, thriftNamespace, typeRegistry);
97  
98          final Document document = context.getDocument();
99          final Header header = document.getHeader();
100         final String javaNamespace = Objects.firstNonNull(swiftGeneratorConfig.getDefaultPackage(), header.getNamespace("java"));
101         Preconditions.checkState(!StringUtils.isEmpty(javaNamespace), "thrift file %s does not declare a java namespace!", thriftFile.getAbsolutePath());
102 
103         for (final String include : header.getIncludes()) {
104             parseDocument(new File(swiftGeneratorConfig.getInputFolder(), include), typeRegistry);
105         }
106 
107         document.visit(new TypeVisitor(javaNamespace, context));
108 
109         return context;
110     }
111 
112     private void generateFiles(final SwiftDocumentContext context) throws IOException
113     {
114         Preconditions.checkState(outputFolder != null, "The output folder was not set!");
115         Preconditions.checkState(outputFolder.isDirectory() && outputFolder.canWrite() && outputFolder.canExecute(), "output folder '%s' is not valid!", outputFolder.getAbsolutePath());
116 
117         final List<DocumentVisitor> visitors = Lists.newArrayList();
118         visitors.add(new ServiceVisitor(templateLoader, context, outputFolder));
119         visitors.add(new StructVisitor(templateLoader, context, outputFolder));
120         visitors.add(new ExceptionVisitor(templateLoader, context, outputFolder));
121         visitors.add(new IntegerEnumVisitor(templateLoader, context, outputFolder));
122         visitors.add(new StringEnumVisitor(templateLoader, context, outputFolder));
123 
124         for (DocumentVisitor visitor : visitors) {
125             context.getDocument().visit(visitor);
126         }
127     }

16  package com.facebook.swift.parser.model;
17  
18  import com.facebook.swift.parser.visitor.DocumentVisitor;
19  import com.google.common.base.Objects;
20  import com.google.common.base.Preconditions;
21  import com.google.common.collect.ImmutableList;
22  
23  import java.io.IOException;
24  import java.util.Collections;
25  import java.util.List;
26  import java.util.Map;
27  
28  import static com.google.common.base.Preconditions.checkNotNull;
29  import static java.util.Collections.emptyList;
30  
31  public class Document
32  {
33      private final Header header;
34      private final List<Definition> definitions;
35  
36      public Document(Header header, List<Definition> definitions)
37      {
38          this.header = checkNotNull(header, "header");
39          this.definitions = ImmutableList.copyOf(checkNotNull(definitions, "definitions"));
40      }
41  
42      public Header getHeader()
43      {
44          return header;
45      }
46  
47      public List<Definition> getDefinitions()
48      {
49          return definitions;
50      }
51  
52      public void visit(final DocumentVisitor visitor) throws IOException
53      {
54          Preconditions.checkNotNull(visitor, "the visitor must not be null!");
55  
56          for (Definition definition : definitions) {
57              if (visitor.accept(definition)) {
58                  definition.visit(visitor);
59              }
60          }
61      }
62  
63  
64      @Override
65      public String toString()
66      {
67          return Objects.toStringHelper(this)
68                  .add("header", header)
69                  .add("definitions", definitions)
70                  .toString();
71      }
72  
73      public static Document emptyDocument()
74      {
75          List<String> includes = emptyList();
76          Map<String, String> namespaces = Collections.emptyMap();
77          List<String> cppIncludes = emptyList();
78          Header header = new Header(includes, namespaces, cppIncludes);
79          List<Definition> definitions = emptyList();
80          return new Document(header, definitions);
81      }
82  }

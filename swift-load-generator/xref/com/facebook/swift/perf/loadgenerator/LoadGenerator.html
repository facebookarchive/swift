40  import com.google.inject.Stage;

42  public class LoadGenerator
43  {
44      private final Provider<AbstractClientWorker> clientWorkerProvider;
45      private final LoadGeneratorCommandLineConfig config;
46      private static Injector injector;
47      private final ThriftClientManager clientManager;
48      private LoadStatsThread loadStatsThread;
49      private AbstractClientWorker[] clientWorkers;
50  
51      public static void main(final String[] args)
52              throws Exception
53      {
54          final LoadGeneratorCommandLineConfig config = new LoadGeneratorCommandLineConfig();
55          JCommander jCommander = new JCommander(config, args);
56  
57          if (config.displayUsage) {
58              jCommander.usage();
59          } else {
60              injector = Guice.createInjector(
61                      Stage.PRODUCTION,
62                      new ConfigurationModule(new ConfigurationFactory(ImmutableMap.<String, String>of())),
63                      new LifeCycleModule(),
64                      new ThriftCodecModule(),
65                      new ThriftClientModule(),
66                      new Module()
67                      {
68                          @Override
69                          public void configure(Binder binder)
70                          {
71                              thriftClientBinder(binder).bindThriftClient(AsyncClientWorker.LoadTest.class);
72                              thriftClientBinder(binder).bindThriftClient(SyncClientWorker.LoadTest.class);
73  
74                              binder.bind(LoadGeneratorCommandLineConfig.class).toInstance(config);
75                              binder.bind(LoadGenerator.class).in(Singleton.class);
76  
77                              if (!config.asyncMode) {
78                                  binder.bind(AbstractClientWorker.class).to(SyncClientWorker.class);
79                              } else {
80                                  binder.bind(AbstractClientWorker.class).to(AsyncClientWorker.class);
81                              }
82                          }
83                      }
84              );
85              injector.getInstance(LifeCycleManager.class).start();
86          }
87      }
88  
89      @Inject
90      public LoadGenerator(LoadGeneratorCommandLineConfig config,
91                           Provider<AbstractClientWorker> clientWorkerProvider,
92                           ThriftClientManager clientManager) {
93          this.config = config;
94          this.clientWorkerProvider = clientWorkerProvider;
95          this.clientManager = clientManager;
96      }
97  
98      @PostConstruct
99      public void start()
100             throws Exception
101     {
102         clientWorkers = new AbstractClientWorker[config.numThreads];
103 
104         for (int i = 0; i < config.numThreads; i++) {
105             clientWorkers[i] = clientWorkerProvider.get();
106         }
107 
108         loadStatsThread = new LoadStatsThread(clientWorkers);
109         loadStatsThread.start();
110 
111         // For fair measurement, start the client workers *after* the monitor has already started
112         for (AbstractClientWorker worker : clientWorkers) {
113             worker.run();
114         }
115     }
116 
117     @PreDestroy
118     public void stop() {
119         for (AbstractClientWorker worker : clientWorkers) {
120             worker.shutdown();
121         }
122         loadStatsThread.shutdown();
123         clientManager.close();
124     }
125 }

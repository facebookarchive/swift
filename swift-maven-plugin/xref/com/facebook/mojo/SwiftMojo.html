39  /**
40   * Process IDL files and generates source code from the IDL files.
41   *
42   * @requiresProject true
43   * @goal generate
44   * @phase generate-sources
45   * @requiresProject true
46   */
47  public class SwiftMojo extends AbstractMojo
48  {
49      private static final Logger LOG = Logger.getLogger(SwiftMojo.class);
50  
51      private static final Function<File, URI> URI_TRANSFORMER = new Function<File, URI>() {
52          @Override
53          public URI apply(final File file) {
54              return file == null ? null : file.toURI();
55          }
56      };
57  
58      /**
59       * Skip the plugin execution.
60       *
61       * @parameter default-value="false"
62       */
63      private boolean skip = false;
64  
65      /**
66       * Override java package for the generated classes. If unset, the java
67       * namespace from the IDL files is used. If a value is set here, the java package
68       * definition from the IDL files is ignored.
69       *
70       * @parameter
71       */
72      private String overridePackage = null;
73  
74      /**
75       * Give a default Java package for generated classes if the IDL files do not
76       * contain a java namespace definition. This package is only used if the IDL files
77       * do not contain a java namespace definition.
78       *
79       * @parameter
80       */
81      private String defaultPackage = null;
82  
83      /**
84       * IDL files to process.
85       *
86       * @parameter
87       * @required
88       */
89      private FileSet idlFiles;
90  
91      /**
92       * Set the Output folder for generated code.
93       *
94       * @parameter default-value="${project.build.directory}/generated-sources/swift"
95       * @required
96       */
97      private File outputFolder = null;
98  
99      /**
100      * Generate code for included IDL files. If true, generate Java code for all IDL files
101      * that are listed in the idlFiles set and all IDL files loaded through include statements.
102      * Default is false (generate only code for explicitly listed IDL files).
103      *
104      * @parameter default-value="false"
105      */
106     private boolean generateIncludedCode = false;
107 
108     /**
109      * Add {@link org.apache.thrift.TException} to each method signature. This exception is thrown
110      * when a thrift internal error occurs.
111      *
112      * @parameter default-value="true"
113      */
114     private boolean addThriftExceptions = true;
115 
116     /**
117      * Have generated services extend {@link Closeable} and a close method.
118      *
119      * @parameter default-value="false"
120      */
121     private boolean addCloseableInterface = false;
122 
123     /**
124      * Generated exceptions extends {@link RuntimeException}, not {@link Exception}.
125      *
126      * @parameter default-value="true"
127      */
128     private boolean extendRuntimeException = true;
129 
130     /**
131      * Select the flavor of the generated source code. Default is "java-regular".
132      *
133      * @parameter default-value="java-regular";
134      */
135     private String codeFlavor = "java-regular";
136 
137     /**
138      * @parameter expression="${project}"
139      * @required
140      * @readonly
141      */
142     protected MavenProject project;
143 
144     @Override
145     public final void execute() throws MojoExecutionException, MojoFailureException
146     {
147         MavenLogAppender.startPluginLog(this);
148 
149         try {
150             if (!skip) {
151 
152                 final File inputFolder = new File(idlFiles.getDirectory());
153 
154                 @SuppressWarnings("unchecked")
155                 final List<File> files = FileUtils.getFiles(inputFolder,
156                                                       Joiner.on(',').join(idlFiles.getIncludes()),
157                                                       Joiner.on(',').join(idlFiles.getExcludes()));
158 
159 
160                 final SwiftGeneratorConfig.Builder configBuilder = SwiftGeneratorConfig.builder()
161                     .inputBase(inputFolder.toURI())
162                     .addInputs(Collections2.transform(files, URI_TRANSFORMER))
163                     .outputFolder(outputFolder)
164                     .overridePackage(overridePackage)
165                     .defaultPackage(defaultPackage)
166                     .generateIncludedCode(generateIncludedCode)
167                     .codeFlavor(codeFlavor);
168 
169                 if (addThriftExceptions) {
170                     configBuilder.addTweak(SwiftGeneratorTweak.ADD_THRIFT_EXCEPTION);
171                 }
172 
173                 if (addCloseableInterface) {
174                     configBuilder.addTweak(SwiftGeneratorTweak.ADD_CLOSEABLE_INTERFACE);
175                 }
176 
177                 if (extendRuntimeException) {
178                     configBuilder.addTweak(SwiftGeneratorTweak.EXTEND_RUNTIME_EXCEPTION);
179                 }
180 
181                 final SwiftGenerator generator = new SwiftGenerator(configBuilder.build());
182                 generator.parse();
183 
184                 project.addCompileSourceRoot(outputFolder.getPath());
185             }
186         }
187         catch (Exception e) {
188             Throwables.propagateIfInstanceOf(e, MojoExecutionException.class);
189             Throwables.propagateIfInstanceOf(e, MojoFailureException.class);
190 
191             LOG.error(String.format("While executing Mojo %s", this.getClass().getSimpleName()), e);
192             throw new MojoExecutionException("Failure:" ,e);
193         }
194         finally {
195             MavenLogAppender.endPluginLog(this);
196         }
197     }

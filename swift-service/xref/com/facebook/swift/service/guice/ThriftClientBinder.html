60      public <T> void bindThriftClient(Class<T> clientInterface)
61      {
62          Preconditions.checkNotNull(clientInterface, "clientInterface is null");
63          String typeName = getServiceName(clientInterface);
64  
65          // Bind ThriftClientConfig with random @Named annotation
66          // We generate a random Named annotation for binding the ThriftClientConfig because we
67          // need one of these for each clientType+annotationType pair.  Without this, the
68          // ThriftClientConfig bindings collapse to a single instance which is shared by all
69          // clients.
70          Named thriftClientConfigKey = Names.named(UUID.randomUUID().toString());
71          bindConfig(binder).annotatedWith(thriftClientConfigKey).prefixedWith(typeName).to(ThriftClientConfig.class);
72  
73          // Bind ThriftClient to a provider which knows how to find the ThriftClientConfig using
74          // the random @Named annotation
75          TypeLiteral<ThriftClient<T>> typeLiteral = toThriftClientTypeLiteral(clientInterface);
76          ThriftClientProviderProvider<T> provider = new ThriftClientProviderProvider<>(clientInterface, DEFAULT_NAME, Key.get(ThriftClientConfig.class, thriftClientConfigKey));
77          binder.bind(typeLiteral).toProvider(provider).in(Scopes.SINGLETON);
78  
79          // Export client to jmx
80          ExportBinder.newExporter(binder)
81                  .export(Key.get(typeLiteral))
82                  .as(format("com.facebook.swift.client:type=%s,clientName=%s",
83                          typeName,
84                          DEFAULT_NAME));
85  
86          // Add the provider itself to a SetBinding so later we can export the thrift methods to JMX
87          Multibinder.newSetBinder(binder, ThriftClientProviderProvider.class).addBinding().toInstance(provider);
88      }

90      public <T> void bindThriftClient(Class<T> clientInterface, Class<? extends Annotation> annotationType)
91      {
92          Preconditions.checkNotNull(clientInterface, "clientInterface is null");
93          String typeName = getServiceName(clientInterface);
94          String name = annotationType.getSimpleName();
95  
96          // Bind ThriftClientConfig with random @Named annotation
97          // see comment on random Named annotation above
98          Named thriftClientConfigKey = Names.named(UUID.randomUUID().toString());
99          String prefix = String.format("%s.%s", typeName, name);
100         bindConfig(binder).annotatedWith(thriftClientConfigKey).prefixedWith(prefix).to(ThriftClientConfig.class);
101 
102         // Bind ThriftClient to a provider which knows how to find the ThriftClientConfig using
103         // the random @Named annotation
104         TypeLiteral<ThriftClient<T>> typeLiteral = toThriftClientTypeLiteral(clientInterface);
105         ThriftClientProviderProvider<T> provider = new ThriftClientProviderProvider<>(clientInterface,
106                 name,
107                 Key.get(ThriftClientConfig.class, thriftClientConfigKey));
108         binder.bind(Key.get(typeLiteral, annotationType)).toProvider(provider).in(Scopes.SINGLETON);
109 
110         // Export client to jmx
111         ExportBinder.newExporter(binder)
112                 .export(Key.get(typeLiteral))
113                 .as(format("com.facebook.swift.client:type=%s,clientName=%s",
114                         typeName,
115                         name));
116 
117         // Add the provider itself to a SetBinding so later we can export the thrift methods to JMX
118         Multibinder.newSetBinder(binder, ThriftClientProviderProvider.class).addBinding().toInstance(provider);
119     }

121     private static String getServiceName(Class<?> clientInterface)
122     {
123         String serviceName = getThriftServiceAnnotation(clientInterface).value();
124         if (!serviceName.isEmpty()) {
125             return serviceName;
126         }
127         return clientInterface.getSimpleName();
128     }

130     /**
131      * @return TypeLiteral<ThriftClient<T>>
132      */
133     private static <T> TypeLiteral<ThriftClient<T>> toThriftClientTypeLiteral(Class<T> clientInterface)
134     {
135         // build a TypeLiteral<ThriftClientProvider<T>> where T is bound to the actual clientInterface class
136         @SuppressWarnings("serial")
137         Type javaType = new TypeToken<ThriftClient<T>>() {}
138                 .where(new TypeParameter<T>() {}, TypeToken.of(clientInterface))
139                 .getType();
140         return (TypeLiteral<ThriftClient<T>>) TypeLiteral.get(javaType);
141     }

143     public static class ThriftClientProviderProvider<T> implements Provider<ThriftClient<T>>
144     {
145         private final Class<T> clientType;
146         private final String clientName;
147         private final Key<ThriftClientConfig> configKey;
148         private ThriftClientManager clientManager;
149         private Injector injector;
150 
151         public ThriftClientProviderProvider(Class<T> clientType, String clientName, Key<ThriftClientConfig> configKey)
152         {
153             Preconditions.checkNotNull(clientType, "clientInterface is null");
154             Preconditions.checkNotNull(clientName, "clientName is null");
155             Preconditions.checkNotNull(configKey, "configKey is null");
156             this.clientType = clientType;
157             this.clientName = clientName;
158             this.configKey = configKey;
159         }
160 
161         @Inject
162         public void setClientManager(ThriftClientManager clientManager)
163         {
164             this.clientManager = clientManager;
165         }
166 
167         @Inject
168         public void setInjector(Injector injector)
169         {
170             this.injector = injector;
171         }
172 
173         @Override
174         public ThriftClient<T> get()
175         {
176             Preconditions.checkState(clientManager != null, "clientManager has not been set");
177             Preconditions.checkState(injector != null, "injector has not been set");
178             ThriftClientConfig clientConfig = injector.getInstance(configKey);
179             return new ThriftClient<>(clientManager, clientType, clientConfig, clientName);
180         }
181 
182         public ThriftClientMetadata getClientMetadata()
183         {
184             Preconditions.checkState(clientManager != null, "clientManager has not been set");
185             return clientManager.getClientMetadata(clientType, clientName);
186         }
187 
188         @Override
189         public boolean equals(Object o)
190         {
191             if (this == o) {
192                 return true;
193             }
194             if (o == null || getClass() != o.getClass()) {
195                 return false;
196             }
197 
198             ThriftClientProviderProvider<?> that = (ThriftClientProviderProvider<?>) o;
199 
200             if (!clientName.equals(that.clientName)) {
201                 return false;
202             }
203             if (!clientType.equals(that.clientType)) {
204                 return false;
205             }
206 
207             return true;
208         }
209 
210         @Override
211         public int hashCode()
212         {
213             int result = clientType.hashCode();
214             result = 31 * result + clientName.hashCode();
215             return result;
216         }
217     }

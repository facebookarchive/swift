57      public ThriftMethodMetadata(Method method, ThriftCatalog catalog)
58      {
59          Preconditions.checkNotNull(method, "method is null");
60          Preconditions.checkNotNull(catalog, "catalog is null");
61  
62          this.method = method;
63  
64          ThriftMethod thriftMethod = method.getAnnotation(ThriftMethod.class);
65          Preconditions.checkArgument(thriftMethod != null, "Method is not annotated with @ThriftMethod");
66  
67          Preconditions.checkArgument(!Modifier.isStatic(method.getModifiers()), "Method %s is static", method.toGenericString());
68  
69          if (thriftMethod.value().length() == 0) {
70              name = method.getName();
71          }
72          else {
73              name = thriftMethod.value();
74          }
75  
76          returnType = catalog.getThriftType(method.getGenericReturnType());
77  
78          ImmutableList.Builder<ThriftFieldMetadata> builder = ImmutableList.builder();
79          Type[] parameterTypes = method.getGenericParameterTypes();
80          String[] parameterNames = extractParameterNames(method);
81          Annotation[][] parameterAnnotations = method.getParameterAnnotations();
82          for (int index = 0; index < parameterTypes.length; index++) {
83              ThriftField thriftField = null;
84              for (Annotation annotation : parameterAnnotations[index]) {
85                  if (annotation instanceof ThriftField) {
86                      thriftField = (ThriftField) annotation;
87                      break;
88                  }
89              }
90  
91              short parameterId = Short.MIN_VALUE;
92              String parameterName = null;
93              if (thriftField != null) {
94                  parameterId = thriftField.value();
95                  if (!thriftField.name().isEmpty()) {
96                      parameterName = thriftField.name();
97                  }
98              }
99              if (parameterId == Short.MIN_VALUE) {
100                 parameterId = (short) (index + 1);
101             }
102             if (parameterName == null) {
103                 parameterName = parameterNames[index];
104             }
105 
106             Type parameterType = parameterTypes[index];
107 
108             ThriftType thriftType = catalog.getThriftType(parameterType);
109 
110             ThriftFieldMetadata fieldMetadata = new ThriftFieldMetadata(
111                     parameterId,
112                     thriftType,
113                     parameterName,
114                     ImmutableList.<ThriftInjection>of(new ThriftParameterInjection(parameterId, parameterName, index, parameterType)),
115                     null,
116                     null
117             );
118             builder.add(fieldMetadata);
119         }
120         parameters = builder.build();
121 
122         exceptions = buildExceptionMap(catalog, thriftMethod);
123 
124         this.oneway = thriftMethod.oneway();
125     }

161     private ImmutableMap<Short, ThriftType> buildExceptionMap(ThriftCatalog catalog,
162                                                               ThriftMethod thriftMethod) {
163         ImmutableMap.Builder<Short, ThriftType> exceptions = ImmutableMap.builder();
164         Set<Type> exceptionTypes = new HashSet<>();
165         int customExceptionCount = 0;
166 
167         if (thriftMethod.exception().length > 0) {
168             for (ThriftException thriftException : thriftMethod.exception()) {
169                 exceptions.put(thriftException.id(), catalog.getThriftType(thriftException.type()));
170                 checkState(exceptionTypes.add(thriftException.type()), "ThriftMethod.exception contains more than one value for %s", thriftException.type());
171             }
172         }
173 
174         for (Class<?> exceptionClass : method.getExceptionTypes()) {
175             if (exceptionClass.isAssignableFrom(TException.class)) {
176                 // the built-in exception types don't need special treatment
177                 continue;
178             }
179 
180             if (exceptionClass.isAnnotationPresent(ThriftStruct.class)) {
181                 ++customExceptionCount;
182 
183                 if (!exceptionTypes.contains(exceptionClass)) {
184                     // there is no rhyme or reason to the order exception types are given to us,
185                     // so we can only infer the id once
186                     checkState(customExceptionCount <= 1, "ThriftMethod.exception annotation value must be specified when more than one custom exception is thrown.");
187                     exceptions.put((short) 1, catalog.getThriftType(exceptionClass));
188                 }
189             }
190         }
191 
192         return exceptions.build();
193     }

195     public boolean isAsync()
196     {
197         Type returnType = method.getGenericReturnType();
198         Class<?> rawType = TypeToken.of(returnType).getRawType();
199         return ListenableFuture.class.isAssignableFrom(rawType);

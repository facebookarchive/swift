16  package com.facebook.swift.service.metadata;

18  import com.facebook.swift.codec.metadata.ThriftCatalog;
19  import com.facebook.swift.service.ThriftMethod;
20  import com.facebook.swift.service.ThriftService;
21  import com.google.common.base.Preconditions;
22  import com.google.common.collect.ImmutableMap;
23  import com.google.common.collect.Iterables;
24  
25  import javax.annotation.concurrent.Immutable;
26  import java.lang.reflect.Method;
27  import java.util.Map;
28  import java.util.Set;
29  
30  import static com.facebook.swift.codec.metadata.ReflectionHelper.findAnnotatedMethods;
31  import static com.facebook.swift.codec.metadata.ReflectionHelper.getAllClassAnnotations;
32  
33  @Immutable
34  public class ThriftServiceMetadata
35  {
36      private final String name;
37      private final Map<String, ThriftMethodMetadata> methods;
38  
39      public ThriftServiceMetadata(Class<?> serviceClass, ThriftCatalog catalog)
40      {
41          Preconditions.checkNotNull(serviceClass, "serviceClass is null");
42          ThriftService thriftService = getThriftServiceAnnotation(serviceClass);
43  
44          if (thriftService.value().length() == 0) {
45              name = serviceClass.getSimpleName();
46          }
47          else {
48              name = thriftService.value();
49          }
50  
51          ImmutableMap.Builder<String, ThriftMethodMetadata> builder = ImmutableMap.builder();
52          for (Method method : findAnnotatedMethods(serviceClass, ThriftMethod.class)) {
53              if (method.isAnnotationPresent(ThriftMethod.class)) {
54                  ThriftMethodMetadata methodMetadata = new ThriftMethodMetadata(method, catalog);
55                  builder.put(methodMetadata.getName(), methodMetadata);
56              }
57          }
58          methods = builder.build();
59      }
60  
61      public ThriftServiceMetadata(String name, ThriftMethodMetadata... methods)
62      {
63          this.name = name;
64  
65          ImmutableMap.Builder<String, ThriftMethodMetadata> builder = ImmutableMap.builder();
66          for (ThriftMethodMetadata method : methods) {
67              builder.put(method.getName(), method);
68          }
69          this.methods = builder.build();
70      }
71  
72      public String getName()
73      {
74          return name;
75      }
76  
77      public ThriftMethodMetadata getMethod(String name)
78      {
79          return methods.get(name);
80      }
81  
82      public Map<String, ThriftMethodMetadata> getMethods()
83      {
84          return methods;
85      }
86  
87      public static ThriftService getThriftServiceAnnotation(Class<?> serviceClass)
88      {
89          Set<ThriftService> serviceAnnotations = getAllClassAnnotations(serviceClass, ThriftService.class);
90          Preconditions.checkArgument(!serviceAnnotations.isEmpty(), "Service class %s is not annotated with @ThriftService", serviceClass.getName());
91          Preconditions.checkArgument(serviceAnnotations.size() == 1,
92                  "Service class %s has multiple conflicting @ThriftService annotations: %s",
93                  serviceClass.getName(),
94                  serviceAnnotations
95          );
96  
97          return Iterables.getOnlyElement(serviceAnnotations);
98      }
99  }
